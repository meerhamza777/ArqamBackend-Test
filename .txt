    // Show logged in user info
    auth.onAuthStateChanged(async (user) => {
      if (user) {
        const uid = user.uid;
        const userDoc = await getDoc(doc(db, "users", uid));
        const name = userDoc.data()?.name;
        userName.textContent = "Welcome, " + name;

        const attendanceDoc = await getDoc(doc(db, "attendance", `${uid}_${today}`));
        if (attendanceDoc.exists()) {
          const data = attendanceDoc.data();
          if (data.checkOutTime) {
            status.textContent = `‚úÖ Checked in at ${data.checkInTime}, out at ${data.checkOutTime}`;
          } else {
            status.textContent = `‚úÖ Checked in at ${data.checkInTime}, not yet checked out`;
          }
        } else {
          status.textContent = "‚ùå Not checked in yet.";
        }

      } else {
        alert("Please login first.");
        window.location.href = "login.html";
      }
    });

    // Check In
    window.checkIn = async () => {
      const user = auth.currentUser;
      const uid = user.uid;
      const time = new Date().toLocaleTimeString();
      const docRef = doc(db, "attendance", `${uid}_${today}`);
      const snap = await getDoc(docRef);

      if (snap.exists()) {
        msg.textContent = "üü° Already checked in.";
      } else {
        await setDoc(docRef, {
          uid,
          date: today,
          checkInTime: time,
          checkOutTime: ""
        });

        status.textContent = `‚úÖ Checked in at ${time}`;

      const uid = user.uid;
      const time = new Date().toLocaleTimeString();
      const docRef = doc(db, "attendance", `${uid}_${today}`);
      const snap = await getDoc(docRef);

        const data = snap.data();

          msg.textContent = "üü† Already checked out.";

          await updateDoc(docRef, { checkOutTime: time });
          msg.textContent = "üîÅ Checked out!";
          status.textContent = `‚úÖ Checked in at ${data.checkInTime}, out at ${time}`;
        }
      } else {
        msg.textContent = "‚ùå Check in first.";
      }
    };
